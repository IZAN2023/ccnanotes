r1:

```bash
ipv6 unicast-routing

interface Loopback0
 no ip address
 ipv6 address 2001:DB8:1:1::10/64
!
interface Ethernet0/0
 no ip address
 ipv6 address 2001:DB8:2:2::1/64
!
ipv6 route 2001:DB8:3:3::/64 2001:DB8:2:2::2
```

ospfv3:

```bash
#r1
ipv6 unicast-routing
!
interface Loopback0
 no ip address
 ipv6 address 2001:DB8:1:1::10/64
 ipv6 ospf 1 area 0
!
interface Ethernet0/0
 no ip address
 ipv6 address 2001:DB8:2:2::1/64
 ipv6 ospf 1 area 0
!
router ospfv3 1
 router-id 1.1.1.1
 !
 address-family ipv6 unicast
 exit-address-family


#r2
ipv6 unicast-routing
!

 ipv6 ospf 1 area 0

!

 ipv6 address 2001:DB8:2:2::2/64
 ipv6 ospf 1 area 0
!

 !
 address-family ipv6 unicast
 exit-address-family
!
```
